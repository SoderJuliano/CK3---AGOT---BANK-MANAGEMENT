agot_bank_dashboard_prepare_projection_effect = {
    set_variable = { name = agot_dash_project_min value = -2 }
    set_variable = { name = agot_dash_project_max value = 4 }

    if = {
        limit = { is_at_war = yes }
        add_to_variable = { name = agot_dash_project_max value = 3 }
        add_to_variable = { name = agot_dash_project_min value = -1 }
    }

    if = {
        limit = { stewardship >= 15 }
        add_to_variable = { name = agot_dash_project_min value = 1 }
        add_to_variable = { name = agot_dash_project_max value = 2 }
    }
    else_if = {
        limit = { stewardship <= 6 }
        add_to_variable = { name = agot_dash_project_min value = -2 }
    }

    if = {
        limit = { diplomacy >= 14 }
        add_to_variable = { name = agot_dash_project_max value = 1 }
    }

    if = {
        limit = { var:agot_dash_bank_interest = -1 }
        add_to_variable = { name = agot_dash_project_min value = 1 }
        add_to_variable = { name = agot_dash_project_max value = 2 }
    }
    else_if = {
        limit = { var:agot_dash_bank_interest = 1 }
        add_to_variable = { name = agot_dash_project_min value = -2 }
        add_to_variable = { name = agot_dash_project_max value = 3 }
    }

    if = {
        limit = { var:agot_dash_bank_risk >= 12 }
        add_to_variable = { name = agot_dash_project_min value = -2 }
        add_to_variable = { name = agot_dash_project_max value = 2 }
    }
}

agot_bank_dashboard_record_monthly_snapshot_effect = {
    if = {
        limit = {
            OR = {
                NOT = { exists = var:agot_dash_last_snapshot_year }
                NOT = { exists = var:agot_dash_last_snapshot_month }
                NOT = { var:agot_dash_last_snapshot_year = current_year }
                NOT = { var:agot_dash_last_snapshot_month = current_month }
            }
        }

        set_variable = { name = agot_dash_history_12 value = var:agot_dash_history_11 }
        set_variable = { name = agot_dash_history_11 value = var:agot_dash_history_10 }
        set_variable = { name = agot_dash_history_10 value = var:agot_dash_history_09 }
        set_variable = { name = agot_dash_history_09 value = var:agot_dash_history_08 }
        set_variable = { name = agot_dash_history_08 value = var:agot_dash_history_07 }
        set_variable = { name = agot_dash_history_07 value = var:agot_dash_history_06 }
        set_variable = { name = agot_dash_history_06 value = var:agot_dash_history_05 }
        set_variable = { name = agot_dash_history_05 value = var:agot_dash_history_04 }
        set_variable = { name = agot_dash_history_04 value = var:agot_dash_history_03 }
        set_variable = { name = agot_dash_history_03 value = var:agot_dash_history_02 }
        set_variable = { name = agot_dash_history_02 value = var:agot_dash_history_01 }

        set_variable = {
            name = agot_dash_temp_den
            value = var:agot_dash_bank_cash
        }
        change_variable = {
            name = agot_dash_temp_den
            add = var:agot_dash_bank_loaned
        }

        if = {
            limit = { var:agot_dash_temp_den > 0 }
            set_variable = {
                name = agot_dash_history_01
                value = var:agot_dash_bank_cash
            }
            change_variable = {
                name = agot_dash_history_01
                multiply = 100
            }
            change_variable = {
                name = agot_dash_history_01
                divide = var:agot_dash_temp_den
            }
        }
        else = {
            set_variable = { name = agot_dash_history_01 value = 0 }
        }

        set_variable = { name = agot_dash_last_snapshot_year value = current_year }
        set_variable = { name = agot_dash_last_snapshot_month value = current_month }

        remove_variable = agot_dash_temp_den
    }
}
