agot_dash_selected_bank_name = {
	type = character

	text = {
		trigger = { var:agot_dash_bank_id = 1 }
		localization_key = agot_dash_bank_name_ib
	}
	text = {
		trigger = { var:agot_dash_bank_id = 2 }
		localization_key = agot_dash_bank_name_bank1
	}
	text = {
		trigger = { var:agot_dash_bank_id = 3 }
		localization_key = agot_dash_bank_name_bank2
	}
	text = {
		trigger = { var:agot_dash_bank_id = 4 }
		localization_key = agot_dash_bank_name_bank3
	}

	text = {
		trigger = {
			NOT = { var:agot_dash_bank_id = 1 }
			NOT = { var:agot_dash_bank_id = 2 }
			NOT = { var:agot_dash_bank_id = 3 }
			NOT = { var:agot_dash_bank_id = 4 }
		}
		localization_key = agot_dash_bank_name_none
	}
}

agot_dash_member_role_text = {
	type = character

	text = {
		trigger = {
			OR = {
				AND = { var:agot_dash_bank_id = 1  has_character_flag = IB_Director }
				AND = { var:agot_dash_bank_id = 2  has_character_flag = bank1_Director }
				AND = { var:agot_dash_bank_id = 3  has_character_flag = bank2_Director }
				AND = { var:agot_dash_bank_id = 4  has_character_flag = bank3_Director }
			}
		}
		localization_key = agot_dash_role_director
	}

	text = {
		trigger = { always = yes }
		localization_key = agot_dash_role_investor
	}
}
